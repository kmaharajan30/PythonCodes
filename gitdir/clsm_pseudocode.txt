	cart_total = cart_total + cart_item
IF Person wish to Remove items
	cart_total = cart_total - cart_item

(Price Discount )
IF cart_total >= 10000    // 10% discount if the total price is above 10000
	cart_total = cart_total - (cart_total * 10/100)

	cart_total = cart_total - (cart_total * 5/100)

(Payment)
READ card
